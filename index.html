number = 73;
isPrime = true;

for(i=2; i<number; i++){
ostatok = number%i;
if(ostatok === 0){
console.log(`${number} % ${i} = ${ostatok}`);
isPrime = false;
console.log(`this is prime: ${isPrime}`);
break
}else{
console.log(`${number} % ${i} = ${ostatok}`);
console.log(`this is prime: ${isPrime}`);
break
}
}

Проверку условия можно разместить под телом цикла, используя специальный синтаксис do..while:
do {
// тело цикла
} while (condition); Цикл сначала выполнит тело, а затем проверит условие condition, и пока его значение равно true, он будет выполняться снова и снова.
Например:
let i = 0;
do {
alert( i );
i++;
} while (i < 3);

Обычно цикл завершается при вычислении условия в false.
Но мы можем выйти из цикла в любой момент с помощью специальной директивы break. Например, следующий код подсчитывает сумму вводимых чисел до тех пор, пока посетитель их вводит, а затем – выдаёт:
let sum = 0;
while (true) {
let value = +prompt("Введите число", '');
if (!value) break; // (*)
sum += value;
}
alert( 'Сумма: ' + sum );
Директива break в строке (*) полностью прекращает выполнение цикла и передаёт управление на строку за его телом, то есть на alert. Вообще, сочетание «бесконечный цикл + break» – отличная штука для тех ситуаций, когда условие, по которому нужно прерваться, находится не в начале или конце цикла, а посередине.

arr - массив, который будет скопирован, arr_2 - новый массив
arr = [
10,
[10,20,30],
[40,50,60],
];

arr_2 = [];

for(i=0; i<arr.length; i++){
if(Array.isArray(arr[i])){
newInnerArray = [];
currentArr = arr[i]; // [10,20,30]

for(j=0; j<currentArr.length; j++){
newInnerArray[newInnerArray.length] = currentArr[j];
}

arr_2[arr_2.length] = newInnerArray;
} else{
arr_2[arr_2.length] = arr[i];
}
}

console.log(arr);
console.log(arr_2);

.flat() в скобках нужно указать число, которое обозначаен на сколько уровненй углублятся для раскладки массива. Если указать знак бесконечности (∞), то массив любого уровня будет разложен в один уровень.
let arr = [
[1,2,3, [10,20,30, [40,50,60]]],
[4,5,6]
];

let arrModified = arr
.flat(Infinity)
.map(function(item){
return item;
})

console.log(arrModified);
