Проверка на простое число

HTML + JS

Функция определяет простое число или нет (в поле number введите число для проверки)
number = 73;
isPrime = true;

for(i=2; i<number; i++){
ostatok = number%i;
if(ostatok === 0){
console.log(`${number} % ${i} = ${ostatok}`);
isPrime = false;
console.log(`this is prime: ${isPrime}`);
break
}else{
console.log(`${number} % ${i} = ${ostatok}`);
console.log(`this is prime: ${isPrime}`);
break
}
}
